
CREATE CONSTRAINT ON (a:Airplane) ASSERT a.Airplane IS UNIQUE;
CREATE CONSTRAINT ON (e:Engine) ASSERT e.Engine IS UNIQUE;
CREATE CONSTRAINT ON (c:Company) ASSERT c.Name IS UNIQUE;

USING PERIODIC COMMIT

LOAD CSV WITH HEADERS FROM 'https://github.com/billjclark/Airplane360/blob/master/Airplanes.csv' AS row
MERGE (airplane:Airplane {Airplane: row.Airplane, model: row.Model})
    ON CREATE SET airplane.name = row.Airplane;
